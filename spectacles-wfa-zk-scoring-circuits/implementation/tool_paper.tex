%%  ABSTRACT
%% =====================================================================
\begin{abstract}
Benchmark evaluations of large language models lack a basic guarantee:
that the reported score was computed correctly.
\spectacles{} produces \emph{score-verified evaluation certificates}---machine-checkable
proofs that a benchmark score matches a formal metric specification.
The key insight is that standard NLP scoring functions (exact match,
BLEU, ROUGE, token~F1) decompose into weighted finite automata (WFA)
over semirings, providing both a formal semantics and a compilation
target for STARK arithmetic circuits.
\spectacles{} compiles these WFAs via a two-tier architecture---algebraic
embedding for $\F_p$-compatible semirings, gadget-assisted compilation
for tropical---and generates zero-knowledge proofs of correct score
computation.

\noindent\textbf{Results.}
We evaluate \spectacles{} on 1{,}015 input pairs across four benchmark
categories (MMLU-style, SQuAD-style, translation, random stress test).
Triple-agreement checks---comparing reference, WFA, and circuit
implementations---achieve \textbf{0~disagreements across 2{,}825 checks}
(100\% agreement).
Compilation correctness testing across 10~random seeds yields
\textbf{57{,}518 total checks with 0~disagreements}, verifying
score-range validity, reflexivity, and edge-case handling (8/8 cases
including empty strings, Unicode, and long inputs).
STARK proofs on simplified circuits (exact match, counting) are
generated in 214\,ms mean proving time with 6.5\,ms mean verification
and ${\sim}44$\,KiB proof sizes, with all 9/9 proofs verified
successfully.
As a secondary contribution, \spectacles{} includes an $n$-gram overlap
detector via OPRF-based private set intersection; this detects
\emph{verbatim} overlap only, not paraphrase memorization.
\end{abstract}

\section{Introduction}\label{sec:intro}

The evaluation of large language models (LLMs) faces a fundamental
trust problem: benchmark scores are computed by imperative scripts
that offer no machine-checkable correctness argument.  A subtle
implementation bug---an off-by-one in $n$-gram clipping, a misplaced
tokenization boundary---can silently inflate or deflate scores, and
no existing infrastructure detects such errors after the
fact~\cite{post2018bleu}.

\paragraph{Why existing approaches fall short.}
VerifiableEvals~\cite{south2024verifiableevals} applies zero-knowledge
proofs to evaluation, but proves that a \emph{specific program} was
executed correctly---it certifies program \emph{execution}, not
specification \emph{conformance}.  If the program contains a bug, the
proof faithfully certifies the buggy answer.

\paragraph{The WFA insight.}
We observe that every standard NLP scoring function decomposes into a
\emph{weighted finite automaton} (WFA) over an appropriate semiring.
Exact match is a Boolean-semiring acceptor; BLEU~\cite{papineni2002bleu}
decomposes into counting-semiring WFAs for $n$-gram counting followed
by aggregation gadgets; ROUGE-L~\cite{lin2004rouge} is a
tropical-semiring WFA computing a longest common subsequence.
This algebraic structure simultaneously provides (i)~a
\emph{formal semantics} rooted in the well-studied theory of weighted
automata~\cite{schutzenberger1961,berstel2011noncommutative}, and
(ii)~a \emph{compilation target} whose transition structure maps
directly to STARK arithmetic constraints.

\paragraph{Scope and honest assessment.}
\spectacles{} addresses \emph{arithmetic correctness}---the problem of
ensuring that a reported score matches its formal specification.  We do
not claim to solve the harder problems of evaluation validity or
specification correctness.  We are explicit about the current
verification boundaries:
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{STARK proofs} are demonstrated on simplified circuits
  (exact match, token counting); full metric circuits (BLEU, ROUGE-L)
  are future work.
\item \textbf{Lean formalization} covers semiring axioms; the
  compilation theorem (EvalSpec $\to$ WFA $\to$ circuit) is empirically
  verified via differential testing, not machine-checked.
\item \textbf{No verified extraction}: there is no verified path from
  Lean specifications to Rust implementation.
\item \textbf{$N$-gram overlap detection} detects verbatim overlap only,
  not paraphrase memorization or indirect contamination.
\end{itemize}

\paragraph{Contributions.}
Within this honestly scoped boundary:
\begin{enumerate}[leftmargin=*,nosep]
\item \textbf{WFA decomposition of NLP metrics}: we show that exact match,
  token~F1, BLEU, ROUGE-1, ROUGE-2, and ROUGE-L decompose into WFAs
  over Boolean, counting, or tropical semirings (\S\ref{sec:wfa}).
\item \textbf{Two-tier compilation}: algebraic embedding for
  $\F_p$-compatible semirings and gadget-assisted compilation for
  tropical (\S\ref{sec:compilation}).
\item \textbf{STARK proof system}: demonstrated on simplified circuits
  with real proof data---214\,ms mean proving, 6.5\,ms mean
  verification, ${\sim}44$\,KiB mean proof size (\S\ref{sec:stark}).
\item \textbf{Comprehensive empirical verification}: 57{,}518 compilation
  checks with 0~disagreements; 2{,}825 triple-agreement checks at 100\%;
  8/8 edge cases pass (\S\ref{sec:eval}).
\item \textbf{EvalSpec DSL}: a typed specification language mapping
  metrics to formal power series over semirings
  (Appendix~\ref{app:evalspec-formal}).
\end{enumerate}

\section{Background}\label{sec:background}

\paragraph{Weighted finite automata.}
A WFA over semiring $\semiring{S}$ and alphabet $\Sigma$ is a tuple
$\wfa = (Q,\Sigma,\{\mu_\sigma\}_{\sigma\in\Sigma},\alpha,\eta)$ where
$\mu_\sigma \in \semiring{S}^{|Q|\times|Q|}$ are transition matrices,
$\alpha \in \semiring{S}^{1\times|Q|}$ is the initial weight vector, and
$\eta \in \semiring{S}^{|Q|\times 1}$ is the final weight vector.  The
weight assigned to input $w = \sigma_1\cdots\sigma_n$ is
$\llbracket\wfa\rrbracket(w) = \alpha \otimes \mu_{\sigma_1} \otimes
\cdots \otimes \mu_{\sigma_n} \otimes
\eta$~\cite{schutzenberger1961,droste2009handbook}.

\paragraph{Semirings.}
A semiring $(\semiring{S},\oplus,\otimes,\bar{0},\bar{1})$ is a set
equipped with two binary operations such that
$(\semiring{S},\oplus,\bar{0})$ is a commutative monoid,
$(\semiring{S},\otimes,\bar{1})$ is a monoid, $\otimes$ distributes
over $\oplus$, and $\bar{0}$ annihilates under $\otimes$.
\spectacles{} uses four semirings: Boolean $(\{0,1\},\lor,\land)$,
counting $(\N,+,\cdot)$, tropical
$(\Z\cup\{+\infty\},\min,+)$, and the Goldilocks field
$\F_p$ ($p=2^{64}-2^{32}+1$).

\paragraph{STARKs.}
A STARK~\cite{bensasson2018stark} enables a prover to convince a verifier
of a computation's correctness without revealing private inputs.  The
computation is expressed as an Algebraic Intermediate Representation
(AIR): polynomial constraints over an execution trace
$\Tr \in \F_p^{T \times w}$.  Soundness relies on
FRI~\cite{bensasson2018fri}, and the system is transparent (no trusted
setup) and post-quantum secure.

\section{WFA Decomposition of NLP Metrics}\label{sec:wfa}

The central theoretical contribution of \spectacles{} is that standard
NLP scoring functions decompose into WFA computations over semirings.
We describe each decomposition and its relationship to Sch\"utzenberger's
characterization of rational formal power
series~\cite{schutzenberger1961}.

\paragraph{Exact match.}
A Boolean-semiring WFA with $|w|+1$ states accepts iff the candidate
matches the reference token-by-token.  State $q_i$ represents ``first
$i$ tokens match''; transition weight is $1$ (match) or $0$ (mismatch).
The entire computation is WFA-expressible.

\paragraph{Token~F1.}
Token~F1 decomposes into two counting-semiring WFAs: one counts
candidate--reference token overlap (precision numerator), the other
counts reference tokens (recall denominator).  The harmonic mean
combining precision and recall falls outside the rational fragment
and is handled by a post-processing gadget.

\paragraph{BLEU.}
BLEU-$n$~\cite{papineni2002bleu} decomposes into $n$ counting-semiring
WFAs, one per $n$-gram order.  Each WFA counts clipped $n$-gram matches
between candidate and reference.  The geometric mean across orders and
the brevity penalty are non-rational operations handled by aggregation
gadgets.

\paragraph{ROUGE-1 and ROUGE-2.}
ROUGE-$n$~\cite{lin2004rouge} uses counting-semiring WFAs for $n$-gram
overlap counting.  The F-measure aggregation is a tested gadget.

\paragraph{ROUGE-L.}
ROUGE-L computes the longest common subsequence (LCS) via a
tropical-semiring $(\Z\cup\{+\infty\},\min,+)$ WFA.  The tropical
semiring is not $\F_p$-embeddable, requiring Tier~2 gadget-assisted
compilation (\S\ref{sec:compilation}).

\paragraph{Expressibility boundary.}
The WFA-expressible fragment corresponds exactly to rational formal
power series (Sch\"utzenberger's theorem).  Operations requiring
non-rational computation---geometric mean, brevity penalty,
$\beta$-weighted F-measure, harmonic mean---are handled by small
aggregation gadgets.  Table~\ref{tab:expressibility} in
Appendix~\ref{app:evalspec-formal} documents which operations are
WFA-expressible and which require post-processing.

%% =====================================================================
\section{Two-Tier Compilation}\label{sec:compilation}

The compiler translates a WFA
$\wfa = (Q,\Sigma,\{\mu_\sigma\},\alpha,\eta)$ over semiring
$\semiring{S}$ into an AIR program over the Goldilocks field
$\F_p$ ($p=2^{64}-2^{32}+1$).

\paragraph{Tier~1: Algebraic compilation.}
When $\semiring{S}$ admits an injective semiring homomorphism
$\phi:\semiring{S}\to\F_p$ (Definition~\ref{def:embedding} in
Appendix~\ref{app:tier1-proof}), the WFA transition
$s_j' = \bigoplus_i s_i \otimes \mu_\sigma[i,j]$ becomes the
field-arithmetic constraint $s_j' = \sum_i s_i \cdot \phi(\mu_\sigma[i,j])$.
The AIR has $2|Q|+O(1)$ columns and one row per input symbol.
Boolean, counting, and real semirings are all
$\F_p$-embeddable (Appendix~\ref{app:bool-embed} addresses the
Boolean $\lor$-idempotency caveat for deterministic WFAs).

\begin{theorem}[Tier~1 soundness]\label{thm:tier1}
Let $\wfa$ be a WFA over an $\F_p$-embeddable semiring~$\semiring{S}$,
and let $\AIR_\wfa$ be the AIR program produced by Tier~1 compilation.
For every input $w\in\Sigma^*$, the unique accepting trace of
$\AIR_\wfa$ on~$w$ encodes the value $\phi(\llbracket\wfa\rrbracket(w))$
in the output column of the final row.
\end{theorem}

\noindent Full proof by induction on input length in
Appendix~\ref{app:tier1-proof}.  \emph{Status}: the theorem statement
is formalized in Lean~4; the compilation pipeline is verified
empirically (57{,}518 checks, 0 disagreements) rather than by
machine-checked proof of the full compilation chain.

\paragraph{Tier~2: Gadget-assisted compilation.}
The tropical semiring $(\Z\cup\{+\infty\},\min,+)$ is not
$\F_p$-embeddable.  Tier~2 introduces \emph{gadgets}---hand-verified
circuit fragments implementing non-algebraic operations.  The comparison
gadget computes $\min(a,b)$ via 64-bit decomposition:
$\min(a,b) = \mathsf{lt}(a,b)\cdot a + (1-\mathsf{lt}(a,b))\cdot b$,
adding 128 auxiliary columns per comparison
(Appendix~\ref{app:tier2-proof}).

\begin{theorem}[Tier~2 soundness for tropical]\label{thm:tier2}
For a tropical-semiring WFA with weight values bounded by $2^{63}$,
Tier~2 compilation correctly encodes
$\llbracket\wfa\rrbracket(w) \bmod p$ in the output column.
\end{theorem}

yet been measured.

\section{$N$-gram Overlap Detection}\label{sec:contamination}

As a secondary contribution, \spectacles{} includes an OPRF-based
private set intersection (PSI) protocol for detecting $n$-gram overlap
between datasets.  We scope this component honestly.

\paragraph{What it detects.}
The PSI protocol computes $|L_1 \cap L_2|$ where $L_1, L_2$ are
$n$-gram sets from two corpora, without revealing set elements.  A trie
optimization exploits prefix sharing, reducing communication from
$O(|L_1|\cdot|L_2|\cdot\lambda)$ to $O((n_1+n_2)\cdot\lambda)$.

\paragraph{What it does not detect.}
This is a \emph{verbatim $n$-gram overlap} check, not a contamination
detector in the general sense.  Specifically:
\begin{enumerate}[nosep]
\item \textbf{Paraphrase memorization}: For $n \geq 4$, even light
  paraphrasing (synonym substitution, clause reordering) reduces overlap
  below any reasonable threshold.  The true-positive rate for paraphrase
  contamination is approximately zero.
\item \textbf{Indirect contamination}: Training on data \emph{derived
  from} benchmark data (e.g., chain-of-thought solutions) produces no
  detectable $n$-gram signal.
\item \textbf{Partial memorization}: Overlap slightly above incidental
  baselines is indistinguishable from natural co-occurrence without
  domain-specific calibration data.
\end{enumerate}
The $n$-gram overlap certificate should be interpreted as evidence
against \emph{verbatim} leakage, not against all forms of data
contamination.

\section{Related Work}\label{sec:related}

\paragraph{Zero-knowledge proof systems.}
STARK~\cite{bensasson2018stark} provides transparent, post-quantum-secure
proofs.  SNARK systems~\cite{groth2016snark} achieve constant-size proofs
but require trusted setup.  Circuit-level DSLs (Circom~\cite{belles2022circom},
Noir~\cite{noir2023}) target R1CS; \spectacles{} targets AIR, better
suited to automaton execution.  EZKL~\cite{kang2024ezkl} proves
neural-network inference, not evaluation metrics.

\paragraph{Verified compilers.}
CompCert~\cite{leroy2009compcert} and CakeML~\cite{kumar2014cakeml}
provide verified compilation with machine-checked proofs and verified
extraction.  \spectacles{} follows a weaker paradigm: verified
\emph{specification} (Lean~4 proofs of semiring axioms) with
\emph{tested} implementation (no verified extraction to Rust).

\paragraph{Verifiable ML evaluation.}
VerifiableEvals~\cite{south2024verifiableevals} certifies program
execution, not specification conformance.  \spectacles{} compiles from
a specification, ensuring correctness by construction for the
WFA-expressible fragment.

\paragraph{Contamination detection.}
Statistical methods~\cite{sainz2023contamination,shi2024contamination}
produce population-level estimates.  \spectacles{}'s $n$-gram overlap
check is complementary but limited to verbatim overlap
(\S\ref{sec:contamination}).

\paragraph{Weighted automata.}
Our WFA decomposition builds on
Sch\"utzenberger~\cite{schutzenberger1961},
Berstel and Reutenauer~\cite{berstel2011noncommutative}, and
Mohri~\cite{mohri2009weighted,droste2009handbook}.

\paragraph{Evaluation frameworks.}
HuggingFace \texttt{evaluate} and TorchMetrics provide widely used
metric implementations but offer no formal correctness guarantees.
\spectacles{} formalizes a subset of these metrics and provides
machine-checkable certificates for the WFA-expressible core.


%% =====================================================================
%%  10.  LIMITATIONS
%% =====================================================================
\section{Limitations}\label{sec:limitations}

We organize limitations by severity.

\paragraph{STARK proofs on simplified circuits only.}
The STARK results (Table~\ref{tab:stark-results}) demonstrate proofs on
exact-match and counting circuits with small state spaces.  Full metric
circuits (BLEU-4 with ${\sim}128$ states, ROUGE-L with
${\sim}1{,}024$ states) have not yet been proved.  We expect
feasibility based on STARK scaling properties, but this is not
demonstrated.

\paragraph{No verified extraction.}
The Lean~4 formalization covers semiring axioms (KleeneSemiring
typeclass) and states compilation soundness theorems.  The Rust
implementation is a separate codebase verified empirically (57{,}518
checks) rather than generated from the Lean specification.  The gap
between specification and implementation is bridged by testing, not
proof.

\paragraph{Post-processing operations are tested, not proved.}
For most metrics, 20--40\% of the computation (geometric mean, harmonic
mean, F-measure, brevity penalty) falls outside the WFA framework.
These gadgets have 0 disagreements across all differential testing but
lack formal soundness proofs.

\paragraph{$N$-gram overlap is not contamination detection.}
As detailed in \S\ref{sec:contamination}, the PSI-based check detects
verbatim $n$-gram overlap only.  It provides no guarantee against
paraphrase memorization, indirect contamination, or semantic leakage.

\paragraph{Protocol composition.}
No UC security proof~\cite{canetti2001uc} exists for the composition of
STARK proofs with PSI.  The commit-then-execute framework provides
concrete defenses but a formal composition theorem is future work.

\paragraph{Metric coverage.}
The supported metrics are string-matching metrics.  Embedding-based
metrics (BERTScore~\cite{zhang2020bertscore}), human preference scores,
and calibration metrics are outside the current WFA framework.


%% =====================================================================
%%  11.  CONCLUSION
%% =====================================================================
\section{Conclusion}\label{sec:conclusion}

\spectacles{} demonstrates that standard NLP scoring functions decompose
into weighted finite automata over semirings, providing a path from
formal metric specifications to zero-knowledge proofs of correct
evaluation.  Our empirical results---57{,}518 compilation checks with
0~disagreements, 2{,}825 triple-agreement checks at 100\%, and 9/9
STARK proofs verified successfully---provide strong evidence for the
correctness of the WFA decomposition and compilation pipeline.

The current system has clear boundaries: STARK proofs are demonstrated
on simplified circuits, the Lean formalization covers axioms but not the
full compilation chain, and the $n$-gram overlap detector handles
verbatim overlap only.  Extending STARK proofs to full metric circuits,
closing the Lean-to-Rust verification gap via verified extraction, and
developing semantic contamination detection are the primary directions
for future work.

\subsection{Typing Rules}

We define the typing judgment $\Gamma \vdash e : (\tau, \semiring{S})$
where $\Gamma$ is a type environment, $\tau$ is a base type, and
$\semiring{S}$ is a semiring annotation.

\begin{gather}
\frac{}{\Gamma \vdash n : (\texttt{Int}, \mathsf{Counting})} \quad (\text{T-IntLit})
\qquad
\frac{}{\Gamma \vdash \texttt{true} : (\texttt{Bool}, \mathsf{Boolean})} \quad (\text{T-BoolLit})
\\[6pt]
\frac{\Gamma(x) = (\tau, \semiring{S})}
     {\Gamma \vdash x : (\tau, \semiring{S})} \quad (\text{T-Var})
\\[6pt]
\frac{\Gamma \vdash e : (\texttt{List}(\texttt{String}), \semiring{S})}
     {\Gamma \vdash \texttt{ngram}(e, n) : (\texttt{List}(\texttt{String}), \mathsf{Counting})}
  \quad (\text{T-NGram})
\\[6pt]
\frac{\Gamma \vdash e_1 : (\tau, \semiring{S}_1) \quad
      \Gamma \vdash e_2 : (\tau, \semiring{S}_2)}
     {\Gamma \vdash e_1 \oplus e_2 : (\tau, \semiring{S}_1 \sqcup \semiring{S}_2)}
  \quad (\text{T-BinOp})
\\[6pt]
\frac{\Gamma \vdash e_1 : (\tau_1, \semiring{S}_1) \quad
      \Gamma[x \mapsto (\tau_1, \semiring{S}_1)] \vdash e_2 : (\tau_2, \semiring{S}_2)}
     {\Gamma \vdash \texttt{let}\;x = e_1\;\texttt{in}\;e_2 : (\tau_2, \semiring{S}_1 \sqcup \semiring{S}_2)}
  \quad (\text{T-Let})
\end{gather}

The semiring lattice orders: $\mathsf{Boolean} \sqsubseteq \mathsf{Counting}
\sqsubseteq \mathsf{Real}$, with $\mathsf{Tropical}$ incomparable.

\subsection{Denotational Semantics}

Every well-typed \evalspec{} expression~$e$ of type $(\tau, \semiring{S})$
denotes a function $\llbracket e \rrbracket : \Sigma^* \times \Sigma^* \to \semiring{S}$:

\begin{align}
\llbracket n \rrbracket_\Gamma &= \lambda(c,r).\; n \\
\llbracket x \rrbracket_\Gamma &= \lambda(c,r).\; \Gamma(x) \\
\llbracket \texttt{ngram}(e, k) \rrbracket_\Gamma &= \lambda(c,r).\;
  \text{extract $k$-grams from } \llbracket e \rrbracket_\Gamma(c,r) \\
\llbracket e_1 \oplus e_2 \rrbracket_\Gamma &= \lambda(c,r).\;
  \llbracket e_1 \rrbracket_\Gamma(c,r) \oplus_{\semiring{S}}
  \llbracket e_2 \rrbracket_\Gamma(c,r) \\
\llbracket \texttt{let}\;x=e_1\;\texttt{in}\;e_2 \rrbracket_\Gamma &=
  \llbracket e_2 \rrbracket_{\Gamma[x \mapsto \llbracket e_1 \rrbracket_\Gamma]}
\end{align}

The compilation correctness theorem states:

\begin{theorem}[Compilation correctness]\label{thm:compile}
For every well-typed \evalspec{} expression~$e$ whose operations are
contained in the rational fragment (i.e., $e$ uses only semiring
operations, $n$-gram extraction, clipping, tokenization, and
composition---excluding aggregation operators such as geometric mean),
the compiler produces a WFA $\wfa_e$ such that
$\llbracket\wfa_e\rrbracket(w) = \llbracket e \rrbracket(w)$
for all inputs~$w \in \Sigma^* \times \Sigma^*$.
\end{theorem}

\noindent\emph{Status}: This theorem is empirically verified (57{,}518
checks, 0 disagreements) but not machine-checked.  The Lean~4
formalization covers the semiring axioms that underpin this theorem;
the full compilation proof is future work.

\subsection{BLEU-4 Example}

metric bleu(n: Int = 4) -> Counting {
  let cand_ngrams = ngram(candidate, n)
  let ref_ngrams  = ngram(reference, n)
  let clipped     = clip(cand_ngrams, ref_ngrams)
  let precision   = clipped / length(cand_ngrams)
  let bp = if length(candidate) < length(reference)
           then exp(1 - length(reference) / length(candidate))
           else 1
  aggregate geometric_mean(
    [precision(1), precision(2), precision(3), precision(4)]
  ) * bp
}
\end{lstlisting}

\subsection{Axiom Verification Methodology}

We verify semiring axioms at two levels:

\paragraph{Lean~4 formalization.}
The \texttt{KleeneSemiring} typeclass axiomatizes commutativity,
associativity, distributivity, identity, and annihilation.  Semiring
axioms for Boolean, counting, and Goldilocks types are machine-checked.
The Kleene star operation is formalized but applied only where the
infinite sum converges (e.g., $0^* = 1$ for counting).

\paragraph{Property-based testing.}
Each axiom is tested via Rust \texttt{proptest} with 256 random inputs
per semiring type.  An independent Python test suite re-verifies on
500 random trials, providing cross-language validation.

\begin{definition}[Embeddable semiring]\label{def:embedding}
A semiring $(\semiring{S},\oplus,\otimes,\bar{0},\bar{1})$ is
\emph{$\F_p$-embeddable} if there exists an injective map
$\phi:\semiring{S}\to\F_p$ satisfying
$\phi(\bar{0}) = 0$, $\phi(\bar{1}) = 1$,
$\phi(a\oplus b) = \phi(a) + \phi(b)$, and
$\phi(a\otimes b) = \phi(a) \cdot \phi(b)$.
\end{definition}

\begin{proposition}[Boolean embedding validity for deterministic WFA]
\label{prop:bool-embed}
In a deterministic WFA, for each state $q$ and input symbol $\sigma$,
at most one outgoing transition has weight~$1$.
Consequently, the inner sum
$\bigoplus_i v_i \otimes \mu_\sigma[i,j] = \bigvee_i (v_i \land \mu_\sigma[i,j])$
has at most one non-zero term, and the $\lor$-idempotency mismatch
($\phi(1 \lor 1) = 1 \neq 2$) never arises.
\end{proposition}

\subsection{Tier~1 Soundness Proof}\label{app:tier1-proof}

\paragraph{Trace layout.}
The execution trace $\Tr\in\F_p^{T\times w}$ has columns:
$\mathbf{s}_0,\ldots,\mathbf{s}_{n-1}$ (current state weights),
$\mathbf{s}_0',\ldots,\mathbf{s}_{n-1}'$ (next state weights),
$\sigma$ (input symbol), and auxiliary bookkeeping columns.

\paragraph{Constraints.}
\begin{align}
  \text{Transition:}\quad s_j' &= \sum_{i=0}^{n-1} s_i \cdot \phi(\mu_\sigma[i,j])
    \label{eq:transition} \\
  \text{Initial boundary:}\quad s_i\big|_{t=0} &= \phi(\alpha_i) \label{eq:boundary-init} \\
  \text{Final boundary:}\quad \text{output} &= \sum_{i=0}^{n-1}
    s_i\big|_{t=T} \cdot \phi(\eta_i) \label{eq:boundary-final}
\end{align}

\begin{proof}[Proof of Theorem~\ref{thm:tier1}]
By induction on $t$.  After processing $\sigma_1\cdots\sigma_t$,
the state vector satisfies $s_i|_t = \phi(v_i^{(t)})$ where
$v^{(t)} = \alpha \otimes \mu_{\sigma_1} \otimes \cdots \otimes \mu_{\sigma_t}$.

\textbf{Base case} ($t=0$): Boundary constraint~\eqref{eq:boundary-init}
sets $s_i|_0 = \phi(\alpha_i) = \phi(v_i^{(0)})$.

\textbf{Inductive step}: By the transition constraint~\eqref{eq:transition}
and the homomorphism property of $\phi$:
\[
  s_j|_{t+1} = \sum_i \phi(v_i^{(t)}) \cdot \phi(\mu_{\sigma_{t+1}}[i,j])
  = \phi\!\left(\bigoplus_i v_i^{(t)} \otimes \mu_{\sigma_{t+1}}[i,j]\right)
  = \phi(v_j^{(t+1)})
\]

\textbf{Final readout}: At $t=T$:
$\text{output} = \sum_i \phi(v_i^{(T)}) \cdot \phi(\eta_i)
= \phi(\llbracket\wfa\rrbracket(w))$.
\end{proof}

\subsection{Tier~2 Soundness Details}\label{app:tier2-proof}

\paragraph{Comparison gadget.}
To compute $\min(a,b)$ over $\F_p$ for $a, b \in [0, 2^{63})$:
\begin{enumerate}[nosep]
\item \textbf{Bit decomposition}: 64 auxiliary columns per operand with
  Boolean constraints $a_i(a_i - 1) = 0$.
\item \textbf{Carry-chain comparison}: Borrow chain from MSB determines
  $\mathsf{lt}(a,b) = 1$ iff $a < b$.
\item \textbf{Min selector}:
  $\min(a,b) = \mathsf{lt}(a,b)\cdot a + (1-\mathsf{lt}(a,b))\cdot b$.
\end{enumerate}

\begin{lemma}[Comparison gadget soundness]\label{lem:comparison}
For $a, b \in [0, 2^{63})$, the comparison gadget correctly computes
$\mathsf{lt}(a,b)$ and $\min(a,b)$ over $\F_p$.
\end{lemma}

\begin{lemma}[Comparison gadget completeness]\label{lem:comparison-complete}
For every $a, b \in [0, 2^{63})$, there exists a unique valid
assignment to the auxiliary columns of the comparison gadget.
\end{lemma}

\subsection{Boolean Embedding}\label{app:bool-embed}

The Boolean semiring $(\{0,1\}, \lor, \land, 0, 1)$ embeds into
$\F_p$ via $\phi(0) = 0$, $\phi(1) = 1$.  The multiplicative
homomorphism holds, but $\lor$ does not map to field addition in general
($\phi(1 \lor 1) = 1 \neq 2$).
Proposition~\ref{prop:bool-embed} shows this mismatch never arises
in deterministic WFAs, where at most one incoming transition carries
weight~1 for any state--symbol pair.
